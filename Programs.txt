
// WAP to check whether a number is prime or not

import java.util.Scanner;
public class PrimeNo {

  public static void main(String[] args) {

    Scanner ob = new Scanner(System.in);
      System.out.println("Enter a number");
    int num = ob.nextInt();
    boolean flag = false;
    for (int i = 2; i <= num / 2; ++i) {
      if (num % i == 0) {
        flag = true;
        break;
      }
    }

    if (!flag){
      System.out.println(num + " is a prime number.");
      }
    else {
      System.out.println(num + " is not a prime number.");
      }
  }
}
 
// WAP to calculate Factorial of a number using recursive function

import java.util.Scanner;
class Examplefactorial{  
 static int factorial(int n){    
  if (n == 0)    
    return 1;    
  else    
    return(n * factorial(n-1));    
 }    
 public static void main(String args[]){  
  int i,fact=1; 
  Scanner ob = new Scanner(System.in);
  System.out.println("Enter the number to calculate factorial ");
  int number = ob.nextInt();
  fact = factorial(number);   
  System.out.println("Factorial of "+number+" is: "+fact);    
 }  
} 

WAP to check if the number is Palindrome or not
import java.util.Scanner;
public class Palindrome {
    public static void main(String[] args) {
        int rem , rev = 0 , temp;
        System.out.println("Enter a number:");
        Scanner ob = new Scanner(System.in);
        int number = ob.nextInt();
        temp = number;
       
        
        while(number>0){
            rem = number%10;
            rev = (rev*10)+rem;
            number = number/10;
            
        }
        
        if(temp==rev){
            System.out.println("Palindrome");
        }
        else{
            System.out.println("Not a palindrome");
        }
         
    } 
}

// WAP to find the second largest number  from an array element.

public class SecondLargest{  
public static int getSecondSmallest(int[] a, int total){  
int temp;  
for (int i = 0; i < total; i++)   
        {  
            for (int j = i + 1; j < total; j++)   
            {  
                if (a[i] > a[j])   
                {  
                    temp = a[i];  
                    a[i] = a[j];  
                    a[j] = temp;  
                }  
            }  
        }  
       return a[1];//2nd element because index starts from 0  
}  
public static void main(String args[]){  
int a[]={1,22,5,6,3,28};    
System.out.println("Second smallest: "+getSecondSmallest(a,6));  
 
}
